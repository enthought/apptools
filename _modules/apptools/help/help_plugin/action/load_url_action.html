[docs]class LoadURLAction(DocAction):
    """ Workbench Action for displaying a url in a web browser.
    """

    def _my_help_doc_default(self):
        exns = self.extension_registry.get_extensions(PARENT +
                                                      '.help_downloads')
        for hc in exns:
            if hc.label == self.name:
                return hc
        return None
